// Variáveis são como "caixas" que guardam dados.
let idade = 25; // 'idade' é considerado um número
let nome = "Ana"; // 'nome' é considerado uma string

// Entrada de dados (simulação com prompt)
idade = parseInt(prompt("Digite sua idade: "));
console.log("Sua idade é:", idade);

if (idade >= 18) { // Se a condição for verdadeira
    console.log("Você é maior de idade.");
} else if (idade >= 16) { // Se a primeira condição não for verdadeira
    console.log("Você pode votar, mas não dirigir.");
} else { // Caso nenhuma condição seja verdadeira
    console.log("Você é menor de idade.");
}

console.log("\nExemplo com for:");
for (let i = 0; i < 5; i++) { // Itera os números: 0, 1, 2, 3, 4
    console.log("O valor de i é:", i);
}

let contador = 0;
while (contador < 5) { // Enquanto a condição for verdadeira
    console.log("O contador é:", contador);
    contador++;
}
