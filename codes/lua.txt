-- Entrada de dados
io.write("Digite sua idade: ")
idade = tonumber(io.read()) -- Converte a entrada para número
print("Sua idade é:", idade)

-- Condicional
if idade >= 18 then -- Se a condição for verdadeira
    print("Você é maior de idade.")
elseif idade >= 16 then -- Se a primeira condição não for verdadeira
    print("Você pode votar, mas não dirigir.")
else -- Caso nenhuma condição seja verdadeira
    print("Você é menor de idade.")
end

-- Exemplo com for
print("\nExemplo com for:")
for i = 0, 4 do -- Itera os números: 0, 1, 2, 3, 4
    print("O valor de i é:", i)
end

-- Exemplo com while
contador = 0
while contador < 5 do -- Enquanto a condição for verdadeira
    print("O contador é:", contador)
    contador = contador + 1
end
