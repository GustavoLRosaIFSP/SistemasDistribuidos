Module Program
    Sub Main()
        Console.WriteLine("Olá, Mundo!") ' Impressão na Tela -> Olá, Mundo!

        Console.WriteLine(2 + 3) ' Impressão na Tela -> 5

        Console.WriteLine("A soma de 2 + 3 é: " & (2 + 3)) ' Impressão -> A soma de 2 + 3 é: 5

        ' Variáveis são como "caixas" que guardam dados.
        Dim idade As Integer = 25 ' 'idade' é considerado um número
        Dim nome As String = "Ana" ' 'nome' é considerado uma string

        ' Entrada de dados
        Console.Write("Digite sua idade: ")
        idade = Integer.Parse(Console.ReadLine()) ' Converte entrada para número
        Console.WriteLine("Sua idade é: " & idade)

        ' Condicional
        If idade >= 18 Then
            Console.WriteLine("Você é maior de idade.")
        ElseIf idade >= 16 Then
            Console.WriteLine("Você pode votar, mas não dirigir.")
        Else
            Console.WriteLine("Você é menor de idade.")
        End If

        ' Exemplo com for
        Console.WriteLine("Exemplo com for:")
        For i As Integer = 0 To 4
            Console.WriteLine("O valor de i é: " & i)
        Next

        ' Exemplo com while
        Dim contador As Integer = 0
        While contador < 5
            Console.WriteLine("O contador é: " & contador)
            contador += 1
        End While
    End Sub
End Module
