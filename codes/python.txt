# input() sempre retorna uma string.
idade = input("Digite sua idade: ")

idade = int(idade)  # Converte para inteiro

    print("Você é maior de idade.")
elif idade >= 16:  # Se a primeira condição não for verdadeira
    print("Você pode votar, mas não dirigir.")
else:  # Caso nenhuma condição seja verdadeira
    print("Você é menor de idade.")
