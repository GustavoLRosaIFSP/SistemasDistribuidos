import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        System.out.println("Olá, Mundo!"); // Impressão na Tela -> Olá, Mundo!

        System.out.println(2 + 3); // Impressão na Tela -> 5

        System.out.println("A soma de 2 + 3 é: " + (2 + 3)); // Impressão na Tela -> A soma de 2 + 3 é: 5

        // Variáveis são como "caixas" que guardam dados.
        int idade = 25; // 'idade' é considerado um int
        String nome = "Ana"; // 'nome' é considerado uma string

        // Entrada de dados
        Scanner scanner = new Scanner(System.in);
        System.out.print("Digite sua idade: ");
        idade = scanner.nextInt();
        System.out.println("Sua idade é: " + idade);

        if (idade >= 18) { // Se a condição for verdadeira
            System.out.println("Você é maior de idade.");
        } else if (idade >= 16) { // Se a primeira condição não for verdadeira
            System.out.println("Você pode votar, mas não dirigir.");
        } else { // Caso nenhuma condição seja verdadeira
            System.out.println("Você é menor de idade.");
        }

        System.out.println("\nExemplo com for:");
        for (int i = 0; i < 5; i++) { // Itera os números: 0, 1, 2, 3, 4
            System.out.println("O valor de i é: " + i);
        }

        int contador = 0;
        while (contador < 5) { // Enquanto a condição for verdadeira
            System.out.println("O contador é: " + contador);
            contador++;
        }

        scanner.close();
    }
}
